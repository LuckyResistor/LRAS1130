89 // Set the LED in the frame register.
90 //
91 inline void setOnOffFrameBit(uint8_t index, uint8_t *data) {
92  const uint8_t segmentIndex = (index/10);
93  const uint8_t segmentLed = (index%10);
94  const uint8_t bitToSet = (1<<(segmentLed&0x7));
95  uint8_t *target = data + (segmentIndex*2);
96  if (segmentLed >= 8) {
97  ++target;
98  }
99  *target |= bitToSet;
100 }

102 // Check if in the definition bit mask a bit at the given coordinates is set.
103 //
104 inline bool isMaskBitSet(uint8_t x, uint8_t y, const uint8_t *data) {
105  const uint8_t *source = data + (y*3) + (x>>3);
106  const uint8_t bitToTest = (1<<(7-(x&7)));
107  return (*source & bitToTest) != 0;
108 }

113 void AS1130::setOnOffFrame24x5(uint8_t frameIndex, const uint8_t *data, uint8_t pwmSetIndex)
114 {
115  // Prepare all frame bytes.
116  const uint8_t finalDataSize = 0x18;
117  uint8_t finalData[finalDataSize];
118  std::memset(finalData, 0, finalDataSize);
119  finalData[1] = (pwmSetIndex<<5);
120  for (uint8_t y = 0; y < 5; ++y) {
121  for (uint8_t x = 0; x < 24; ++x) {
122  const uint8_t ledIndex = y + (5 * x);
123  if (isMaskBitSet(x, y, data)) {
124  setOnOffFrameBit(ledIndex, finalData);
125  }
126  }
127  }
128  // Write the bytes
129  const uint8_t frameAddress = (RS_OnOffFrame + frameIndex);
130  for (uint8_t i = 0; i < finalDataSize; ++i) {
131  writeToMemory(frameAddress, i, finalData[i]);
132  }
133 }

136 void AS1130::setOnOffFrameAllOn(uint8_t frameIndex, uint8_t pwmSetIndex)
137 {
138  const uint8_t frameAddress = (RS_OnOffFrame + frameIndex);
139  // Write the first segment with the PWM set index.
140  writeToMemory(frameAddress, 0, 0xff);
141  writeToMemory(frameAddress, 1, (pwmSetIndex<<5)|0x03);
142  // Write all other segments
143  for (uint8_t i = 1; i < 12; ++i) {
144  writeToMemory(frameAddress, i*2, 0xff);
145  writeToMemory(frameAddress, i*2+1, 0x07);
146  }
147 }

150 void AS1130::setBlinkAndPwmSetAll(uint8_t setIndex, bool doesBlink, uint8_t pwmValue)
151 {
152  const uint8_t setAddress = (RS_BlinkAndPwmSet + setIndex);
153  // Disable all blink flags.
154  for (uint8_t i = 0; i < 12; ++i) {
155  if (doesBlink) {
156  writeToMemory(setAddress, i*2, 0xff);
157  writeToMemory(setAddress, i*2+1, 0x07);
158  } else {
159  writeToMemory(setAddress, i*2, 0x00);
160  writeToMemory(setAddress, i*2+1, 0x00);
161  }
162  }
163  // Set all PWM values to the maximum.
164  for (uint8_t i = 0x18; i < 0x9c; ++i) {
165  writeToMemory(setAddress, i, pwmValue);
166  }
167 }

376 AS1130::LedStatus AS1130::getLedStatus(uint8_t ledIndex)
377 {
378  if (ledIndex > 0xba) {
379  return LedStatusDisabled;
380  }
381  if ((ledIndex & 0x0f) > 0xa) {
382  return LedStatusDisabled;
383  }
384  const uint8_t ledBitMask = (1<<(ledIndex&0x7));
385  const uint8_t registerIndex = CR_OpenLedBase + (ledIndex>>3);
386  const uint8_t mask = readFromMemory(RS_Control, registerIndex);
387  if ((mask & ledBitMask) == 0) {
388  return LedStatusOpen;
389  } else {
390  return LedStatusOk;
391  }
392 }

439 {
