28 void setup() {
29  // Initialize the Wire library.
30  Wire.begin();
31 
32  // Initialize the serial port.
33  Serial.begin(9600);
34 
35  // Wait until the chip is ready.
36  delay(100);
37  Serial.println(F("Initialize chip"));
38 
39  // Check if the chip is addressable.
40  if (!ledDriver.isChipConnected()) {
41  Serial.println(F("Communication problem with chip."));
42  Serial.flush();
43  return;
44  }
45 
46  // Set-up everything.
47  ledDriver.setRamConfiguration(AS1130::RamConfiguration1);
48  ledDriver.setOnOffFrameAllOn(0);
49  ledDriver.setBlinkAndPwmSetAll(0);
50  ledDriver.setCurrentSource(AS1130::Current30mA);
51  ledDriver.setScanLimit(AS1130::ScanLimitFull);
52  ledDriver.startPicture(0);
53 
54  // Enable the chip
55  ledDriver.startChip();
56 
57  // Check for dead LEDs
58  Serial.println(F("Run the LED test"));
59  ledDriver.runManualTest();
60 
61  // Display the status of all leds.
62  for (uint8_t ledIndex = 0x00; ledIndex < 0xbb; ++ledIndex) {
63  Serial.print(F("LED 0x"));
64  Serial.print(ledIndex, HEX);
65  Serial.print(F(": "));
66  switch (ledDriver.getLedStatus(ledIndex)) {
67  case AS1130::LedStatusOk:
68  Serial.println(F(" OK"));
69  break;
70  case AS1130::LedStatusOpen:
71  Serial.println(F(" open"));
72  break;
73  case AS1130::LedStatusDisabled:
74  Serial.println(F(" disabled"));
75  break;
76  }
77  }
78 }
